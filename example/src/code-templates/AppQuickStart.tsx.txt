import { dialog, dialogStore } from '@/lib/dialogs';
import { DialogsRenderer } from 'react-layered-dialog';

export function App() {
  const showConfirm = () => {
    dialog.confirm({
      title: '안내',
      message: '새로운 DialogStore 기반 API를 사용 중입니다.',
    });
  };

  return (
    <div className="flex min-h-screen flex-col items-center justify-center gap-4">
      <button className="rounded bg-black px-4 py-2 text-white" onClick={showConfirm}>
        확인 열기
      </button>

      <DialogsRenderer store={dialogStore} />
    </div>
  );
}
