import { useDialogs } from '@/lib/dialogs';
import type { DialogState } from 'react-layered-dialog';
import type { AlertDialogState } from '@/lib/dialogs';
import { Button } from '@/components/ui/button';

type AlertProps = DialogState<AlertDialogState>;

export const Alert = ({
  id,
  title,
  message,
  onOk,
  zIndex,
  closeOnOutsideClick = true,
  dimmed = true,
}: AlertProps) => {
  const { closeDialog } = useDialogs();

  const handleClose = () => {
    onOk?.();
    closeDialog(id);
  };

  return (
    <div
      className={`fixed inset-0 flex items-center justify-center ${
        dimmed ? 'bg-black/40 pointer-events-auto' : 'bg-transparent pointer-events-none'
      }`}
      style={{ zIndex }}
      role="alertdialog"
      aria-modal="true"
      aria-labelledby="alert-title"
      aria-describedby="alert-message"
      onClick={closeOnOutsideClick ? handleClose : undefined}
    >
      <div
        className="pointer-events-auto relative min-w-[280px] rounded-lg bg-white p-6 shadow-xl"
        onClick={(event) => event.stopPropagation()}
      >
        <h3 id="alert-title" className="text-lg font-semibold">
          {title}
        </h3>
        <p id="alert-message" className="mt-2 text-sm text-muted-foreground">
          {message}
        </p>
        <div className="mt-4 flex justify-end">
          <Button autoFocus onClick={handleClose}>
            확인
          </Button>
        </div>
      </div>
    </div>
  );
};
